[docs]    def __init__(self, parent=None):
        """
        Создадим нужные компоненты и произведём настройку внешнего вида окна.
        """
        def append_properties():
            """
            Загрузим форму и добави её к свойствам объекта.
            """
            form, base = uic.loadUiType("first_face.ui")
            self.ui = form()
            self.ui.setupUi(self)


        def restore_state():
            """
            Здесь восстанавливаем настройки графического интерфейса.
            """
            settings = self._settings()
            self.restoreGeometry(settings.value('/first_face/geometry').toByteArray())
            self.restoreState(settings.value('/first_face/state').toByteArray())


        def create_tree_model():
            from stone.stone import FlyingStone
            stone = FlyingStone()
            stone.init({'mass': 100, 'V0': 134, 'angle': 44})
            stone.prepare_for_stepping()
            model = Model(self.ui.properties_editor)
            model.set_model(stone)
            self.ui.properties_editor.setModel(model)


        QtGui.QMainWindow.__init__(self, parent)
        append_properties()
        create_tree_model()
        restore_state()

        def save_state():
            """
            Здесь сохраняем настройки графического интерфейса: размер окна, выбранные меню и т.п.

            settings = self._settings()

        save_state()
