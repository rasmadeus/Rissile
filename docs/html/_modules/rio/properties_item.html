        add_part()

[docs]    def show(self, column, role):
        """
        Функция предназначена для вызова из QAbstractItemModel, для отображения свойства на экране.
        Путём реализации соответствующих функций, вы можете настроить вид каждого конкретного свойства.

        showman = {
            QtCore.Qt.DisplayRole: self._display_role,
            QtCore.Qt.FontRole: self._font_role
        }
        return showman.get(role, lambda column: QtCore.QVariant())(column)

[docs]    def change_type(self, type, init_value=0.0):
        """
        Функция меняет части self в зависимости от типа.
        """
        def create_parts_for_const_generator():
            """
            Меняет части self в соответсвии с исходными данными для rissile.wo.data_generator.ValueConstGenerator
            """
            ValueEditable('value', init_value, self)
            from rissile.wo.data_generator import ValueConstGenerator
            return ValueConstGenerator


        def create_parts_for_range_generator():
            """
            Меняет части self в соответсвии с исходными данными для rissile.wo.data_generator.ValueRangeGenerator
            """
            ValueEditable('begin', init_value, self)
            ValueEditable('right_border', init_value, self)
            ValueEditable('step', init_value, self)
            from rissile.wo.data_generator import ValueRangeGenerator
            return ValueRangeGenerator


        def create_parts_for_random_generator():
            """
            Меняет части self в соответсвии с исходными данными для rissile.wo.data_generator.ValueRandomGenerator
            """
            ValueEditable('left_border', init_value, self)
            ValueEditable('right_border', init_value, self)
            from rissile.wo.data_generator import ValueRandomGenerator
            return ValueRandomGenerator


        makers = {\
            'const generator': create_parts_for_const_generator,\
            'range generator': create_parts_for_range_generator,\
            'random generator': create_parts_for_random_generator\
        }


        self.clear_parts()
        self._generator = makers.get(type, create_parts_for_const_generator)()

[docs]    def fill(self, wo_init_data_generator):
        """
        Заполняет генератор значением, соответствующему self.

        :type wo_init_data_generator: rissile.wo._data_generator.WorldObjectTestDataGenerator
        """
        def params():
            """
            :return: параметры для инициализации генератора.
            """
            params = {}
            for part in self._parts:
                params[part._name] = part._value
            return params

        
        wo_init_data_generator.set_value(self._create_generator(self._name, params()))

[docs]    def restore_from_params(self, params):
        """
        Функция выстраивает иерархию с вершиной self в соответствии с данными params.

        :type params: dict
        """
        for id, value in params.iteritems():
            if isinstance(value, dict):
                part = Root(id, self)
                part.restore_from_params(value)
            else:
                ValueForGenerator(id, 'const generator', self)




[docs]    def generator(self):
        """
        Функция возвращает генератор исходных данных, в соответствии с иерархией self.
        """
        from rissile.wo.data_generator import ParamsGenerator
        wo_init_data_generator = ParamsGenerator(self._name)
        for part in self._parts:
            part.fill(wo_init_data_generator)
        return wo_init_data_generator
