[docs]def about_zero(value, tolerance=TOLERANCE_VALUE):
    """
    :return: истину, если переданное значение равно нулю с заданной точностью.

    >>> first_angle = 3.0 - 2.7
    >>> second_angle = 0.3
    >>> print(first_angle)
    0.3
    >>> print(second_angle)
    0.3
    >>> first_angle == second_angle
    False
    >>> about_zero(first_angle - second_angle)
    True
    """
    import math
    return math.fabs(value) <= tolerance
